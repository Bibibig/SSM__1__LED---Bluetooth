#include <SoftwareSerial.h>


SoftwareSerial BTSerial(5, 6); //Connect HC-06 TX,RX
//RX ÇÉ 6¹ø ²ÈŸ»
void setup() 
{
  Serial.begin(9600);
  Serial.println("Hello!");

  // set the data rate for the BT port
  BTSerial.begin(9600);
}

void loop()
{
  if (BTSerial.available())
    Serial.write(BTSerial.read());
  if (Serial.available())
    BTSerial.write(Serial.read());
}
